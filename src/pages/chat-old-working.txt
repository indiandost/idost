        method: "POST",

      imageUrl = data.url;

      from: Number(myId),
      to: Number(friendId),
      message,
      media_url: imageUrl,
      type: imageUrl ? "image" : "text",

const startAgoraCall = async (type) => {
  setCallLoading(true);
  if (clientRef.current) {
    console.log("⚠️ Already in call");
    return;
  }

  try {

    const channel = [myId, friendId].sort().join("_");

    const rtcClient = AgoraRTC.createClient({
      mode: "rtc",
      codec: "vp8",
    });

    await rtcClient.join(APP_ID, channel, null, Number(myId));
    await rtcClient.enableAudioVolumeIndicator();

   clientRef.current = rtcClient;
   setClient(rtcClient);

    // 🔥 clear old video safely
    const localDiv = document.getElementById("local-video");
    const remoteDiv = document.getElementById("remote-video");

    if (localDiv) localDiv.replaceChildren();
    if (remoteDiv) remoteDiv.replaceChildren();

    let mic = null;
    let cam = null;

    // =============================
    // 🎤 CREATE MIC
    // =============================
    try {
      mic = await AgoraRTC.createMicrophoneAudioTrack();
    } catch (err) {
      console.log("⚠️ No microphone available");
    }

    // =============================
    // 🎥 CREATE CAMERA (VIDEO CALL)
    // =============================
    if (type === "video") {
      try {

        cam = await AgoraRTC.createCameraVideoTrack();

        if (localDiv) {
          setTimeout(() => {
            localDiv.replaceChildren();
            cam.play(localDiv);
          }, 300);
        }

      } catch (err) {
        console.log("⚠️ Camera unavailable");
      }
    }

    // =============================
    // 🚀 PUBLISH AVAILABLE TRACKS
    // =============================
  const tracks = [mic, cam].filter(Boolean);

if (tracks.length > 0) {
  await rtcClient.publish(tracks);

  // ✅ force enable
  if (mic) await mic.setEnabled(true);
  if (cam) await cam.setEnabled(true);
}

// ✅ hide loader after publish success
setCallLoading(false);

    // ✅ save tracks
    setLocalTracks({ mic, cam });

    // =============================
    // 👤 REMOTE USER HANDLER
    // =============================
   rtcClient.on("user-published", async (user, mediaType) => {

  if (user.uid == myId) return;

  await rtcClient.subscribe(user, mediaType);

  // 🎥 REMOTE VIDEO
  if (mediaType === "video" && user.videoTrack) {

    const remoteDiv = document.getElementById("remote-video");

    if (remoteDiv) {

      remoteDiv.replaceChildren();

      setTimeout(() => {
        try {
          user.videoTrack.play(remoteDiv);
        } catch (err) {
          console.log("Remote video play error", err);
        }
      }, 500);
    }
  }

  // 🎤 REMOTE AUDIO
 /* if (mediaType === "audio" && user.audioTrack) {

    setTimeout(() => {
      try {
        user.audioTrack.play();
      } catch (err) {
        console.log("Remote audio play error", err);
      }
    }, 500);
  }*/
  if (mediaType === "audio" && user.audioTrack) {
    try {
      user.audioTrack.play();
      console.log("🔊 Remote audio playing");
    } catch (err) {
      console.log("Remote audio play error", err);
    }
  }
});

    // =============================
    // 👋 REMOTE USER LEFT
    // =============================
    rtcClient.on("user-unpublished", () => {
      if (remoteDiv) {
        remoteDiv.replaceChildren();
      }
    });

  } catch (err) {
    setCallLoading(false); 
    console.error("Agora error:", err);
    alert("❌ Unable to start call");

const endCall = async () => {

  const duration = callStartRef.current
    ? Math.floor((Date.now() - callStartRef.current) / 1000)
    : 0;

  socket.emit("sendMessage", {
    from: Number(myId),
    to: Number(friendId),
    message: `${callTypeRef.current} call duration ${duration}s`,
    type: "call_accepted",
    createdAt: new Date().toISOString(),
  });

  socket.emit("endCall", {
    from: myId,
    to: friendId,
  });

  await cleanupCall();
};
